CREATE DATABASE flaskapp;
USE flaskapp;
CREATE TABLE users(
	id INT NOT NULL AUTO_INCREMENT,
	firstName VARCHAR(30) NOT NULL, 
    	lastName VARCHAR(30) NOT NULL, 
    	phone VARCHAR(30) NOT NULL,
	PRIMARY KEY (id)
);

CLEAN TABLE DATA:
TRUNCATE TABLE flaskapp.users;

SHOW PROCESSLIST;
